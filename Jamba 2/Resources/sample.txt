Here's a clean implementation:
python
def process_paragraphs(
    paragraphs: list[str],
    tokenize: callable,        # your existing function: str -> list[int] (padded to 1024)
    max_len: int = 1024,
    pad_token: int = 0
) -> list[list[int]]:
    """
    Accepts a list of paragraph strings.
    Returns a list of token arrays, each exactly max_len integers.
    - Short paragraphs are padded to max_len.
    - Long paragraphs are split into multiple max_len chunks in-place.
    """
    result = []

    for paragraph in paragraphs:
        # Get raw tokens (strip padding first if tokenize() always pads)
        padded = tokenize(paragraph)
        tokens = _strip_padding(padded, pad_token)

        # Split into chunks of max_len
        chunks = _chunk_tokens(tokens, max_len)

        # Pad each chunk to exactly max_len
        for chunk in chunks:
            padded_chunk = chunk + [pad_token] * (max_len - len(chunk))
            result.append(padded_chunk)

    return result


def _strip_padding(tokens: list[int], pad_token: int) -> list[int]:
    """Remove trailing pad tokens from a padded sequence."""
    # Strip from the right
    end = len(tokens)
    while end > 0 and tokens[end - 1] == pad_token:
        end -= 1
    return tokens[:end]


def _chunk_tokens(tokens: list[int], max_len: int) -> list[list[int]]:
    """Split a flat token list into chunks of at most max_len."""
    if not tokens:
        return [[]]  # Empty paragraph still produces one (padded) chunk
    return [tokens[i:i + max_len] for i in range(0, len(tokens), max_len)]
How it works:
_strip_padding — since your existing tokenize() always pads to 1024, we first strip the trailing pad tokens to get the raw token sequence before rechunking.
_chunk_tokens — splits the raw tokens into slices of max_len. A 2500-token paragraph produces three chunks: [1024, 1024, 452].
Back in process_paragraphs — each chunk is re-padded to exactly max_len, then appended to the result. The output list can be longer than the input list when paragraphs overflow.
Example:
python
# Input: 3 paragraphs
paragraphs = [
    "short text",          # → 1 chunk  (padded)
    "very long text ...",  # → 3 chunks (split + last padded)
    "medium text ...",     # → 2 chunks (split + last padded)
]

output = process_paragraphs(paragraphs, tokenize)
# len(output) == 6, each output[i] has exactly 1024 ints
One edge case to be aware of: _strip_padding strips trailing pad tokens. If your tokenizer uses pad_token=0 but 0 is also a valid content token (e.g., an <unk> id), you may want to use a sentinel pad id that doesn't collide with real tokens — or have your tokenizer return the unpadded length alongside the tokens.

Here's the C++ implementation:
cpp
#include <vector>
#include <string>
#include <functional>

using TokenArray = std::vector<int>;
using TokenMatrix = std::vector<TokenArray>;

// Your existing function signature (assumed)
// TokenArray tokenize(const std::string& text);  // returns padded vector of 1024 ints

static TokenArray strip_padding(const TokenArray& tokens, int pad_token = 0) {
    int end = static_cast<int>(tokens.size());
    while (end > 0 && tokens[end - 1] == pad_token) {
        --end;
    }
    return TokenArray(tokens.begin(), tokens.begin() + end);
}

TokenMatrix process_paragraphs(
    const std::vector<std::string>& paragraphs,
    const std::function<TokenArray(const std::string&)>& tokenize,
    int max_len = 1024,
    int pad_token = 0
) {
    TokenMatrix result;

    for (const auto& paragraph : paragraphs) {
        // Tokenize and strip padding to get raw tokens
        TokenArray raw = strip_padding(tokenize(paragraph), pad_token);

        // Empty paragraph still produces one padded chunk
        if (raw.empty()) {
            result.push_back(TokenArray(max_len, pad_token));
            continue;
        }

        // Split into chunks of max_len and pad each one
        for (int i = 0; i < static_cast<int>(raw.size()); i += max_len) {
            TokenArray chunk(raw.begin() + i,
                             raw.begin() + std::min(i + max_len,
                                                    static_cast<int>(raw.size())));
            // Pad to exactly max_len
            chunk.resize(max_len, pad_token);
            result.push_back(std::move(chunk));
        }
    }

    return result;
}

So the new approach is simpler — just tokenize the raw text once (no bucketing), then split into 1024-sized chunks. Here's the rewrite:
cpp
TokenMatrix tokenize(const std::string& text, int pad_token = gpadtoken_id) {
    const uint32_t length = 1024;
    TokenMatrix result;

    if (!gchunker || text.empty()) {
        result.push_back(TokenArray(length, pad_token));
        return result;
    }

    TokenArray raw;
    try {
        auto tokens = gchunker->tokenize(text);
        for (auto t : tokens)
            raw.push_back((int)t);
    } catch (const std::exception& e) {
        std::fprintf(stderr, "ERROR tokenize failed: %s\n", e.what());
        result.push_back(TokenArray(length, pad_token));
        return result;
    }

    if (raw.empty()) {
        result.push_back(TokenArray(length, pad_token));
        return result;
    }

    for (size_t i = 0; i < raw.size(); i += length) {
        TokenArray chunk(raw.begin() + i,
                         raw.begin() + std::min(i + (size_t)length, raw.size()));
        chunk.resize(length, pad_token);
        result.push_back(std::move(chunk));
    }

    return result;
}
Key changes:
gchunker->tokenize(text) instead of gchunker->chunk() — no bucketing, no ChunkParams
length is now a constant 1024 rather than a parameter, since bucketing is gone
pad_token defaults to gpadtoken_id directly

Two problems:
1. Default argument gpadtoken_id in the signature — default arguments are evaluated at compile time, but gpadtoken_id is a global set at runtime. This is undefined behavior and likely resolves to 0 or garbage. Fix by using a sentinel:
cpp
static PA_CollectionRef tokenize(const std::string& text,
                                 const uint32_t length = 1024,
                                 int pad_token = -1) {
    if (pad_token == -1) pad_token = gpadtoken_id;
2. process_paragraphs appends the whole chunks collection as a single element — ob_append_c wraps it as one nested element rather than flattening the chunks in. You need to iterate and append each chunk individually:
cpp
PA_CollectionRef process_paragraphs(
    const std::vector<std::string>& paragraphs,
    const uint32_t length
) {
    PA_CollectionRef result = PA_CreateCollection();
    for (const auto& paragraph : paragraphs) {
        PA_CollectionRef chunks = tokenize(paragraph, length);
        long n = PA_GetCollectionLength(chunks);
        for (long i = 0; i < n; i++) {
            PA_Variable v = PA_GetCollectionElement(chunks, i);
            PA_SetCollectionElement(result, PA_GetCollectionLength(result), v);
        }
    }
    return result;
}

You want the last chunk to have tokens first, then padding — which it already does. But you're saying the first chunk is showing filler tokens prepended, meaning PA_CreateCollection is not starting empty.
The fix is to not use PA_CreateCollection for raw and the chunks at all for the token-copying path — instead, build a plain std::vector<int> internally and only convert to PA_CollectionRef at the very end when appending:
cpp
static PA_CollectionRef tokenize(const std::string& text,
                                 const uint32_t length = 1024,
                                 int pad_token = -1) {
    if (pad_token == -1) pad_token = gpadtoken_id;

    PA_CollectionRef result = PA_CreateCollection();

    auto append_chunk = [&](const std::vector<int>& chunk) {
        PA_CollectionRef c = PA_CreateCollection();
        for (int t : chunk)
            ob_append_n(c, t);
        ob_append_c(result, c);
    };

    if (!gchunker || text.empty()) {
        append_chunk(std::vector<int>(length, pad_token));
        return result;
    }

    std::vector<int> raw;
    try {
        auto tokens = gchunker->tokenize(text);
        for (auto t : tokens)
            raw.push_back((int)t);
    } catch (const std::exception& e) {
        std::fprintf(stderr, "ERROR tokenize failed: %s\n", e.what());
        append_chunk(std::vector<int>(length, pad_token));
        return result;
    }

    if (raw.empty()) {
        append_chunk(std::vector<int>(length, pad_token));
        return result;
    }

    for (size_t i = 0; i < raw.size(); i += length) {
        std::vector<int> chunk(raw.begin() + i,
                               raw.begin() + std::min(i + (size_t)length, raw.size()));
        chunk.resize(length, pad_token);  // only pads the last chunk
        append_chunk(chunk);
    }

    return result;
}
By keeping everything in std::vector<int> until the final append_chunk call, we completely sidestep whatever PA_CreateCollection is doing with its initial state. The proprietary collections are only used for output.

Yes, just check if it's the last iteration:
cpp
for (PA_long32 i = 0; i < raw_size; i += length) {
    PA_CollectionRef chunk = PA_CreateCollection();
    long end = std::min(i + (long)length, raw_size);
    bool is_last = (end == raw_size);

    if (is_last) {
        // pre-fill with pad tokens, then overwrite with real tokens at the end
        for (long j = 0; j < (long)length; j++)
            ob_append_n(chunk, pad_token);
        long token_count = end - i;
        long pad_count = length - token_count;
        for (PA_long32 j = i; j < end; j++) {
            PA_Variable v = PA_GetCollectionElement(raw, j);
            if (PA_GetVariableKind(v) == eVK_Real)
                PA_SetCollectionElement(chunk, pad_count + (j - i), v);
        }
    } else {
        for (PA_long32 j = i; j < end; j++) {
            PA_Variable v = PA_GetCollectionElement(raw, j);
            if (PA_GetVariableKind(v) == eVK_Real)
                ob_append_n(chunk, PA_GetRealVariable(v));
        }
    }

    ob_append_c(result, chunk);
}
For the last chunk it fills the collection with length pad tokens first, then uses PA_SetCollectionElement to overwrite the tail end with the real tokens — giving you [pad, pad, ..., t0, t1, ..., tn]. All other chunks are written normally token by token.
